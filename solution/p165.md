# Intersections

## Description du Problème
Un générateur pseudo-aléatoire (Blum Blum Shub) génère des segments de ligne. Chaque segment est défini par quatre nombres générés : `(x0, y0, x1, y1)`.

**Objectif** : Combien de points d'intersection distincts (vrais) sont générés par 5000 segments de ligne ?

---

## Exemples
### Exemple 1
**Entrée** : 2 segments
**Sortie** : Nombre de points d'intersection distincts
**Explication** : Si deux segments se croisent en un point, ce point est compté une fois.

---

## Approche de Résolution

### Stratégie
1. Générer 5000 segments de ligne en utilisant le générateur Blum Blum Shub.
2. Pour chaque paire de segments (i, j) avec i < j :
   - Calculer le point d'intersection en utilisant la formule d'intersection de deux segments.
   - Vérifier si l'intersection est à l'intérieur des deux segments (0 < t0 < 1 et 0 < t1 < 1).
   - Si oui, ajouter le point à un ensemble (pour éviter les doublons).
3. Utiliser des fractions pour représenter les points avec précision exacte.
4. Le générateur BBS : `state = (state * state) % 50515093`, puis `return state % 500`.

### Observations
- On doit utiliser des fractions pour éviter les erreurs d'arrondi.
- La formule d'intersection de deux segments nécessite de vérifier que les paramètres sont dans [0, 1].

### Complexité
- **Temporelle** : O(n²), où n=5000.
- **Spatiale** : O(n²) pour stocker les points d'intersection.

---

## Solution en C++

```cpp
#include <iostream>
#include <set>
#include <vector>
#include <algorithm>

using namespace std;

struct Fraction {
    long long num, den;
    
    Fraction(long long n = 0, long long d = 1) {
        long long g = __gcd(n, d);
        num = n / g;
        den = d / g;
        if (den < 0) {
            num = -num;
            den = -den;
        }
    }
    
    bool operator<(const Fraction& other) const {
        return num * other.den < other.num * den;
    }
    
    bool operator==(const Fraction& other) const {
        return num == other.num && den == other.den;
    }
    
    Fraction operator+(const Fraction& other) const {
        return Fraction(num * other.den + other.num * den, den * other.den);
    }
    
    Fraction operator*(const Fraction& other) const {
        return Fraction(num * other.num, den * other.den);
    }
};

struct Point {
    Fraction x, y;
    
    bool operator<(const Point& other) const {
        if (x < other.x) return true;
        if (other.x < x) return false;
        return y < other.y;
    }
};

struct Segment {
    int x0, y0, x1, y1;
};

class BbsRandom {
private:
    int state;
    
public:
    BbsRandom() : state(290797) {}
    
    int next() {
        state = (int)((long long)state * state % 50515093);
        return state % 500;
    }
};

int main() {
    const int NUM_SEGMENTS = 5000;
    BbsRandom rand;
    vector<Segment> segments;
    
    for (int i = 0; i < NUM_SEGMENTS; i++) {
        Segment seg;
        seg.x0 = rand.next();
        seg.y0 = rand.next();
        seg.x1 = rand.next();
        seg.y1 = rand.next();
        segments.push_back(seg);
    }
    
    set<Point> intersections;
    Fraction FRAC_ONE(1, 1);
    Fraction FRAC_ZERO(0, 1);
    
    for (int i = 0; i < segments.size(); i++) {
        Segment seg0 = segments[i];
        for (int j = i + 1; j < segments.size(); j++) {
            Segment seg1 = segments[j];
            
            int x0 = seg0.x0, y0 = seg0.y0;
            int x1 = seg0.x1, y1 = seg0.y1;
            int x2 = seg1.x0, y2 = seg1.y0;
            int x3 = seg1.x1, y3 = seg1.y1;
            
            int denom = (x0 - x1) * (y2 - y3) - (x2 - x3) * (y0 - y1);
            if (denom == 0) continue;
            
            int numer0 = (x0 - x2) * (y2 - y3) - (x2 - x3) * (y0 - y2);
            int numer1 = (x1 - x0) * (y0 - y2) - (x0 - x2) * (y1 - y0);
            
            Fraction t0(numer0, denom);
            Fraction t1(numer1, denom);
            
            if (FRAC_ZERO < t0 && t0 < FRAC_ONE && FRAC_ZERO < t1 && t1 < FRAC_ONE) {
                Fraction x = Fraction(x0, 1) + t0 * Fraction(x1 - x0, 1);
                Fraction y = Fraction(y0, 1) + t0 * Fraction(y1 - y0, 1);
                intersections.insert({x, y});
            }
        }
    }
    
    cout << intersections.size() << endl;
    return 0;
}
```
