# Équation de Pell

## Description du Problème
Considérons les équations diophantiennes quadratiques de la forme :
x² - Dy² = 1

Par exemple, quand D = 13, la solution minimale en x est 649² - 13 × 180² = 1.

On peut supposer qu'il n'y a pas de solutions en entiers positifs quand D est un carré.

En trouvant les solutions minimales en x pour D = {2, 3, 5, 6, 7}, on obtient :
- 3² - 2 × 2² = 1
- 2² - 3 × 1² = 1
- 9² - 5 × 4² = 1
- 5² - 6 × 2² = 1
- 8² - 7 × 3² = 1

Par conséquent, en considérant les solutions minimales en x pour D ≤ 7, le plus grand x est obtenu quand D = 5.

**Objectif** : Trouver la valeur de D ≤ 1000 dans les solutions minimales de x pour laquelle la plus grande valeur de x est obtenue.

---

## Exemples
### Exemple 1
**Entrée** : D ≤ 7
**Sortie** : `5`
**Explication** : Pour D = 5, la solution minimale est x = 9, qui est la plus grande parmi D ≤ 7.

---

## Approche de Résolution

### Stratégie
1. Pour chaque D de 2 à 1000 (en excluant les carrés parfaits), trouver la solution minimale en x.
2. Utiliser le développement en fraction continue de √D pour trouver la solution fondamentale de l'équation de Pell.
3. Si la période de la fraction continue est paire, la solution est le convergent avant la fin de la période.
4. Si la période est impaire, utiliser une formule spéciale basée sur le convergent.
5. Trouver le D qui donne le plus grand x.

### Observations
- L'équation de Pell x² - Dy² = 1 a toujours une solution fondamentale.
- La solution peut être trouvée en utilisant le développement en fraction continue de √D.
- On utilise une classe QuadraticSurd similaire au problème 64.

### Complexité
- **Temporelle** : O(D × P), où D est la limite (1000) et P est la période moyenne.
- **Spatiale** : O(P), pour stocker les termes de la fraction continue.

---

## Solution en C++

```cpp
#include <iostream>
#include <vector>
#include <map>
#include <gmpxx.h>
#include <cmath>
#include <algorithm>

using namespace std;

// Classe QuadraticSurd similaire au problème 64
class QuadraticSurd {
public:
    mpz_class a, b, c, d;
    
    QuadraticSurd(mpz_class a, mpz_class b, mpz_class c, mpz_class d) {
        if (c == 0) throw invalid_argument("c cannot be zero");
        if (c < 0) { a = -a; b = -b; c = -c; }
        mpz_class g = gcd(gcd(a, b), c);
        if (g != 1) { a /= g; b /= g; c /= g; }
        this->a = a; this->b = b; this->c = c; this->d = d;
    }
    
    QuadraticSurd subtract(QuadraticSurd other) {
        if (d != other.d) throw invalid_argument("d must match");
        return QuadraticSurd(a*other.c - other.a*c, b*other.c - other.b*c, c*other.c, d);
    }
    
    QuadraticSurd reciprocal() {
        return QuadraticSurd(-a*c, b*c, b*b*d - a*a, d);
    }
    
    mpz_class floor() {
        mpz_class temp = sqrt(b*b*d);
        if (b < 0) temp = -(temp + 1);
        temp += a;
        if (temp < 0) temp -= (c - 1);
        return temp / c;
    }
    
    bool operator<(const QuadraticSurd& other) const {
        if (a != other.a) return a < other.a;
        if (b != other.b) return b < other.b;
        if (c != other.c) return c < other.c;
        return d < other.d;
    }
    
private:
    mpz_class gcd(mpz_class x, mpz_class y) {
        while (y != 0) { mpz_class t = y; y = x % y; x = t; }
        return x;
    }
};

struct Fraction {
    mpz_class numerator, denominator;
    Fraction(mpz_class n) : numerator(n), denominator(1) {}
    Fraction(mpz_class n, mpz_class d) : numerator(n), denominator(d) {}
    Fraction add(Fraction other) {
        return Fraction(numerator * other.denominator + other.numerator * denominator,
                       denominator * other.denominator);
    }
};

vector<vector<mpz_class>> sqrtToContinuedFraction(int n) {
    vector<mpz_class> terms;
    map<QuadraticSurd, int> seen;
    QuadraticSurd val(0, 1, 1, n);
    int startPeriod = -1;
    do {
        if (seen.find(val) != seen.end()) {
            startPeriod = seen[val];
            break;
        }
        seen[val] = terms.size();
        mpz_class flr = val.floor();
        terms.push_back(flr);
        val = val.subtract(QuadraticSurd(flr, 0, 1, val.d)).reciprocal();
    } while (true);
    
    vector<mpz_class> prefix(terms.begin(), terms.begin() + startPeriod);
    vector<mpz_class> period(terms.begin() + startPeriod, terms.end() - 1);
    return {prefix, period};
}

mpz_class smallestSolutionX(int n) {
    vector<vector<mpz_class>> contFrac = sqrtToContinuedFraction(n);
    vector<mpz_class> temp;
    temp.insert(temp.end(), contFrac[0].begin(), contFrac[0].end());
    temp.insert(temp.end(), contFrac[1].begin(), contFrac[1].end());
    
    Fraction val(temp.back());
    for (int i = temp.size() - 2; i >= 0; i--) {
        val = Fraction(val.denominator, val.numerator).add(Fraction(temp[i]));
    }
    
    if (contFrac[1].size() % 2 == 0)
        return val.numerator;
    else
        return val.numerator * val.numerator + val.denominator * val.denominator * n;
}

int main() {
    int minN = -1;
    mpz_class maxX = 0;
    
    for (int n = 2; n <= 1000; n++) {
        int sqrt_n = (int)sqrt(n);
        if (sqrt_n * sqrt_n == n)
            continue;
        mpz_class x = smallestSolutionX(n);
        if (x > maxX) {
            minN = n;
            maxX = x;
        }
    }
    
    cout << minN << endl;
    return 0;
}
```
